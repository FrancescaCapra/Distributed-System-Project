111    public abstract static class Node extends AbstractActor {
112      protected int id; // node ID
113      protected int v;  // value replica
114      protected boolean isCrashed = false;  // track crash state
115      protected List<ActorRef> participants;      // list of participant nodes
116      protected Decision decision = null;         // decision taken by this node

118      public Node(int id, int v) {
119        super();

127      void setGroup(StartMessage sm) {
128        participants = new ArrayList<>();
129        for (ActorRef b: sm.group) {
130          if (!b.equals(getSelf())) {
131  
132            // copying all participant refs except for self
133            this.participants.add(b);
134          }
135        }
136        print("starting with " + sm.group.size() + " peer(s)");
137      }

238    public static void main(String[] args) {
239  
240      // Create the actor system
241      final ActorSystem system = ActorSystem.create("helloakka");
242  
243      // Create the coordinator
244      ActorRef coordinator = system.actorOf(Coordinator.props(), "coordinator");
245  
246      // Create participants
247      List<ActorRef> replicas = new ArrayList<>();
248      for (int i = 0; i< N_REPLICAS; i++) {
249        replicas.add(system.actorOf(Replica.props(i, -1), "participant" + i));
250      }
251  
252      // Create participants
253      List<ActorRef> clients = new ArrayList<>();
254      for (int i = 0; i< N_CLIENT; i++) {
255        clients.add(system.actorOf(Client.props(i, -1), "client" + i));
256      }
257  
258      // Send start messages to the replicas to inform them of the replicas
259      StartMessage start = new StartMessage(replicas);
260      for (ActorRef peer: replicas) {
261        peer.tell(start, null);
262      }
263  
264      // Send start messages to the clients to inform them of the replicas
265      //  StartMessage startclient = new StartMessage(clients);
266      for (ActorRef peer: clients) {
267        peer.tell(start, null);
268      }
269  
270      // Send the start messages to the coordinator
271      coordinator.tell(start, null);
272  
273  
274      /*clients.forEach(client -> {

277
