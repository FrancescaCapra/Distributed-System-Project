17   public class Coordinator extends Node {
18   
19       // here all the nodes that sent YES are collected
20       private final Set<ActorRef> yesVoters = new HashSet<>();
21       private Set<ActorRef> ackVoters = new HashSet<>();
22       private int pendingValue = 0;  // This holds the value pending confirmation
23       private boolean writeConfirmed = false;  // To track if the write confirmation has been sent
24       private Cancellable heartbeatTask;  // To manage the heartbeat sending task
25   
26       boolean allVotedYes() { // returns true if all voted YES
27           return yesVoters.size() >= N_REPLICAS;
28       }
29   
30       public Coordinator() {
31           super(-1, -1); // the coordinator has the id -1
32       }
33   
34       static public Props props() {
35           return Props.create(Coordinator.class, Coordinator::new);
36       }
37   
38       @Override
39       public Receive createReceive() {
40           return receiveBuilder()
41                   .match(StartMessage.class, this::onStartMessage)
42                   .match(UpdateRequest.class, this::onUpdateRequest)
43                   .match(Acknowledge.class, this::onAcknowledge)
44                   .match(VoteResponse.class, this::onVoteResponse)
45                   .match(Timeout.class, this::onTimeout)
46                   .match(DecisionRequest.class, this::onDecisionRequest)
47                   .build();
48       }
49   
50       // This method sets up a task to send heartbeat messages
51       private void startHeartbeat() {
52           heartbeatTask = getContext().getSystem().scheduler().scheduleAtFixedRate(
53                   Duration.create(5, TimeUnit.SECONDS), // Start after 5 seconds
54                   Duration.create(5, TimeUnit.SECONDS), // Repeat every 5 seconds
55                   this::sendHeartbeat,
56                   getContext().getSystem().dispatcher()
57           );
58       }
59   
60       // Sends a heartbeat to all participants except itself
61       private void sendHeartbeat() {
62           for (ActorRef participant : participants) {
63               if (!participant.equals(getSelf())) {
64                   participant.tell(new Heartbeat(), getSelf());
65               }
66           }
67       }
68   
69       @Override
70       public void preStart() {
71           startHeartbeat();  // Start sending heartbeats when the actor starts
72       }
73   
74       public void onStartMessage(StartMessage msg) {                   /* Start */
75           setGroup(msg);
76           print("Sending vote request");
77           multicast(new VoteRequest());
78           //multicastAndCrash(new VoteRequest(), 3000);
79           setTimeout(VOTE_TIMEOUT);
80           //crash(5000);
81       }
82   
83       public void onUpdateRequest(UpdateRequest msg) {
84           pendingValue = msg.newValue;
85           ackVoters.clear();
86           multicast(new UpdateMessage(msg.newValue));
87           setTimeout(VOTE_TIMEOUT);
88       }
89   
90       public void onAcknowledge(Acknowledge msg) {
91           ackVoters.add(getSender());
92           if (!writeConfirmed && ackVoters.size() >= (N_REPLICAS / 2) + 1) {
93               // Iterate over the participants and send the message only to replicas
94               for (ActorRef participant : participants) {
95                   if (!participant.equals(getSelf())) {  // Check to ensure not sending to self
96                       participant.tell(new WriteOk(), getSelf());
97                   }
98               }
99               writeConfirmed = true;  // Set the flag to prevent further messages
100              print("Write operation confirmed. Value updated to " + pendingValue);
101          }
102      }
103  
104  
105      public void onVoteResponse(VoteResponse msg) {                    /* Vote */
106          if (hasDecided()) {
107  
108              // we have already decided and sent the decision to the group,
109              // so do not care about other votes
110              return;
111          }
112          Vote v = (msg).vote;
113          if (v == Vote.YES) {
114              yesVoters.add(getSender());
115              if (allVotedYes()) {
116                  fixDecision(Decision.COMMIT);
117                  //if (id==-1) {crash(3000); return;}
118                  multicast(new DecisionResponse(decision));
119                  //multicastAndCrash(new DecisionResponse(decision), 3000);
120              }
121          }
122          else { // a NO vote
123  
124              // on a single NO we decide ABORT
125              fixDecision(Decision.ABORT);
126              multicast(new DecisionResponse(decision));
127          }
128      }
129  
130      public void onTimeout(Timeout msg) {
131          if (!hasDecided()) {
132              print("Timeout");
133  
134              // not decided in time means ABORT
135              fixDecision(Decision.ABORT);
136              multicast(new DecisionResponse(Decision.ABORT));
137          }
138      }
