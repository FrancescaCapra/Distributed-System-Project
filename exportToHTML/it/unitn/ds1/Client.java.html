12   /*-- Client -----------------------------------------------------------*/
13   public class Client extends Node {
14       public Client(int id, int v) { super(id, v); }
15   
16       static public Props props(int id, int v) {
17           return Props.create(Client.class, () -> new Client(id, v));
18       }
19   
20       public void onStartMessage(StartMessage msg) {
21           //System.out.printf("ciao\n\n");
22           // replica[random]
23           setGroup(msg);
24       }
25   
26       public void requestRead(ReadValueRequest msg) {
27           int selectedRandReplica = new Random().nextInt(participants.size());  // replica selezionata per effettuare operazione di read o write.
28           ActorRef selectedReplica = participants.get(selectedRandReplica);
29           selectedReplica.tell(new ReadValueRequest(), getSelf());
30           print("Requested read from replica " + selectedReplica);
31       }
32   
33       public void requestWrite(int newValue) {
34           int selectedReplicaIndex = new Random().nextInt(participants.size());
35           ActorRef selectedReplica = participants.get(selectedReplicaIndex);
36           selectedReplica.tell(new WriteValueRequest(newValue), getSelf());
37           print("Write request sent to replica " + selectedReplicaIndex + " with new value: " + newValue);
38       }
39   
40   
41       public void onTimeout(Timeout msg) {
42           if (!hasDecided()) {
43               print("Timeout. Replica has crashed.");
44   
45               // ask other participants
46               multicast(new DecisionRequest());
47   
48               // ask also the coordinator
49               //coordinator.tell(new DecisionRequest(), getSelf());
50               //setTimeout(DECISION_TIMEOUT);
51           }
52       }
53   
54   
55   
56       @Override
57       public Receive createReceive() {
58           return receiveBuilder()
59                   .match(StartMessage.class, this::onStartMessage)
60                   .match(ReadValueRequest.class, this::requestRead)
61                   .match(WriteValueRequest.class, req -> requestWrite(req.newValue))
62                   .match(Timeout.class, this::onTimeout)
63                   .build();
64       }
